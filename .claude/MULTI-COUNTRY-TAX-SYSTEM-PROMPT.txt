IMPLEMENT MULTI-COUNTRY TAX DECLARATION & BANK DETAILS SYSTEM

OBJECTIVE: Create a flexible, scalable tax declaration and bank details system that handles multiple countries
with different tax identification requirements (TFN for Australia, PAN for India/Nepal, NRIC for Singapore, etc.)
and different bank detail formats (BSB for Australia, IFSC for India/Nepal, IBAN for Europe, etc.).

REFERENCE:
- Current tax info schema: /lib/db/schemas/employee-tax-info.ts
- Current bank details schema: /lib/db/schemas/employee-bank-details.ts
- Current tax validation: /lib/validations/employee-payroll.ts
- Tax form: /components/employees/forms/TaxInfoForm.tsx
- Bank form: /components/employees/forms/BankDetailsForm.tsx
- API route: /app/api/employees/[id]/tax-info/route.ts

import { z } from 'zod'
import { CountryCode, getCountryConfig } from '@/lib/config/countries'

// Tax ID validation patterns
const TAX_ID_PATTERNS: Record<string, RegExp> = {
  tfn: /^\d{11}$/, // 11 digits (without hyphens)
  pan: /^[A-Z]{5}[0-9]{4}[A-Z]{1}$/, // 10 chars: 5 letters, 4 digits, 1 letter
  nric: /^[SFG]\d{7}[A-Z]$/, // Singapore format
  ird: /^\d{8,9}$/, // 8 or 9 digits
  tax_code: /^[0-9]{1,4}[A-Z]{0,2}$/, // UK tax code
  ssn: /^\d{9}$/, // 9 digits
  sin: /^\d{9}$/, // 9 digits
}

const BANK_ID_PATTERNS: Record<string, RegExp> = {
  bsb: /^\d{6}$/, // 6 digits (without hyphen)
  ifsc: /^[A-Z]{4}0[A-Z0-9]{6}$/, // IFSC code
  iban: /^[A-Z]{2}\d{2}[A-Z0-9]{1,30}$/, // IBAN format
  swift_code: /^[A-Z]{4}[A-Z]{2}[A-Z0-9]{2}([A-Z0-9]{3})?$/, // SWIFT code
  routing: /^\d{9}$/, // US routing number
  transit: /^\d{5}$/, // Canada transit number
}

// Base schema for all countries
const baseTaxDeclarationSchema = z.object({
  taxIdType: z.string(),
  taxId: z.string().min(1, 'Tax ID is required'),
  currency: z.string().optional(),
  dateUpdated: z.date().default(() => new Date()),
})

// Australia-specific schema
const australiaTaxSchema = baseTaxDeclarationSchema.extend({
  taxId: z.string()
    .regex(TAX_ID_PATTERNS.tfn, 'TFN must be 11 digits')
    .transform(v => v.replace(/\D/g, '')), // Remove non-digits
  taxFreeThreshold: z.boolean().default(false),
  seniorTaxOffset: z.boolean().default(false),
  zoneOverseasCarer: z.boolean().default(false),
  studentLoan: z.boolean().default(false),
  financialSupplementDebt: z.boolean().default(false),
  superannuationFund: z.string().optional(),
  superannuationMemberNumber: z.string().optional(),
  salarySecureFormUrl: z.string().url().optional(),
  salarySecureFormSignedDate: z.date().optional(),
})

// India/Nepal-specific schema
const indiaNepaliTaxSchema = baseTaxDeclarationSchema.extend({
  taxId: z.string()
    .regex(TAX_ID_PATTERNS.pan, 'PAN must be 10 characters (e.g., AAAPA5055K)')
    .toUpperCase(),
  taxIdName: z.string().min(1, 'PAN cardholder name is required'),
  uanNumber: z.string().optional(), // UAN for EPF
  citizenship: z.string().optional(), // For Nepal
  panIssueDate: z.date().optional(),
})

// UK-specific schema
const ukTaxSchema = baseTaxDeclarationSchema.extend({
  taxCode: z.string().min(1, 'Tax code is required'),
  studentLoan: z.boolean().default(false),
  ukTaxYearStatus: z.enum(['standard', 'emergency', 'non_resident']).default('standard'),
})

// Singapore-specific schema
const singaporeTaxSchema = baseTaxDeclarationSchema.extend({
  taxId: z.string()
    .regex(TAX_ID_PATTERNS.nric, 'NRIC/FIN must be in format S/FN1234567A')
    .toUpperCase(),
  nricType: z.enum(['citizen', 'permanent_resident', 'foreign_worker']),
  legalStatus: z.string(),
  nationality: z.string(),
  race: z.string(),
  religion: z.string(),
  maritalStatus: z.enum(['single', 'married', 'widowed', 'divorced']),
  issueDate: z.date().optional(), // Required for PR and FW
  expiryDate: z.date().optional(), // Required for FW
})

// New Zealand-specific schema
const newzealandTaxSchema = baseTaxDeclarationSchema.extend({
  taxId: z.string()
    .regex(TAX_ID_PATTERNS.ird, 'IRD must be 8 or 9 digits'),
  nzTaxCode: z.string().min(1, 'NZ tax code is required'),
  nzEsctRate: z.number().min(0).max(100).default(0),
  nzSourceOfIncome: z.string().optional(),
  nzVariedTaxRate: z.number().optional(), // Required when nzTaxCode is WT, STC, or STC_SL
})

// US-specific schema
const usTaxSchema = baseTaxDeclarationSchema.extend({
  taxId: z.string()
    .regex(TAX_ID_PATTERNS.ssn, 'SSN must be 9 digits')
    .transform(v => v.replace(/\D/g, '')),
  w4FilingStatus: z.enum(['single', 'married_jointly', 'married_separately', 'head_of_household']),
})

// Canada-specific schema
const canadaTaxSchema = baseTaxDeclarationSchema.extend({
  taxId: z.string()
    .regex(TAX_ID_PATTERNS.sin, 'SIN must be 9 digits')
    .transform(v => v.replace(/\D/g, '')),
  province: z.string().min(2, 'Province is required'),
  td1OnFile: z.boolean().optional(),
})

// Master function to get schema for country
export function getTaxDeclarationSchema(country: CountryCode) {
  switch (country) {
    case 'australia':
      return australiaTaxSchema
    case 'india':
    case 'nepal':
      return indiaNepaliTaxSchema
    case 'uk':
      return ukTaxSchema
    case 'singapore':
      return singaporeTaxSchema
    case 'newzealand':
      return newzealandTaxSchema
    case 'us':
      return usTaxSchema
    case 'canada':
      return canadaTaxSchema
    default:
      return baseTaxDeclarationSchema
  }
}

export type AustraliaTaxData = z.infer<typeof australiaTaxSchema>
export type IndiaNepaliTaxData = z.infer<typeof indiaNepaliTaxSchema>
export type UKTaxData = z.infer<typeof ukTaxSchema>
export type SingaporeTaxData = z.infer<typeof singaporeTaxSchema>
export type NewZealandTaxData = z.infer<typeof newzealandTaxSchema>
export type USTaxData = z.infer<typeof usTaxSchema>
export type CanadaTaxData = z.infer<typeof canadaTaxSchema>

import { CountryCode, getCountryConfig } from '@/lib/config/countries'

export function maskTaxId(country: CountryCode, value: string): string {
  if (!value) return '***'

  switch (country) {
    case 'australia':
      // TFN: XXX-XXX-123 (show last 3)
      return `***-***-${value.slice(-3)}`

    case 'india':
    case 'nepal':
      // PAN: AA****K (show first 2 and last 1)
      return `${value.slice(0, 2).toUpperCase()}****${value.slice(-1).toUpperCase()}`

    case 'singapore':
      // NRIC: S****67A (show first and last 3)
      return `${value.slice(0, 1)}****${value.slice(-3).toUpperCase()}`

    case 'uk':
      // Tax Code: 125*L (show first and last)
      return `${value.slice(0, 1)}****${value.slice(-1)}`

    case 'newzealand':
      // IRD: ****5678 (show last 4)
      return `****${value.slice(-4)}`

    case 'us':
      // SSN: ***-**-6789 (show last 4)
      return `***-**-${value.slice(-4)}`

    case 'canada':
      // SIN: ***-***-789 (show last 3)
      return `***-***-${value.slice(-3)}`

    default:
      return '***'
  }
}

export function maskBankId(country: CountryCode, idType: string, value: string): string {
  if (!value) return '****'

  switch (country) {
    case 'australia':
      if (idType === 'bsb') {
        // BSB: 123-***
        return `${value.slice(0, 3)}-***`
      }
      break

    case 'india':
    case 'nepal':
      if (idType === 'ifsc') {
        // IFSC: SBIN****
        return `${value.slice(0, 4)}****`
      }
      break

    case 'uk':
    case 'newzealand':
      if (idType === 'iban') {
        // IBAN: GB82 WEST ****
        return `${value.slice(0, 6)}****`
      }
      break
  }

  return '****'
}

export function maskAccountNumber(accountNumber: string): string {
  if (!accountNumber || accountNumber.length < 4) return '••••'
  return `••••${accountNumber.slice(-4)}`
}

import { toast } from 'sonner'
import { CountryCode, getCountryConfig } from '@/lib/config/countries'
import { getTaxDeclarationSchema } from '@/lib/validations/tax-declaration'
import { updateEmployeeTaxInfo } from '@/lib/api/employees'
import { AustraliaTaxForm } from './tax-declaration/AustraliaTaxForm'
import { IndiaNepaliTaxForm } from './tax-declaration/IndiaNepaliTaxForm'
import { UKTaxForm } from './tax-declaration/UKTaxForm'
import { SingaporeTaxForm } from './tax-declaration/SingaporeTaxForm'
import { NewZealandTaxForm } from './tax-declaration/NewZealandTaxForm'

interface TaxDeclarationFormProps {
  employeeId: string
  country: CountryCode
  isEditing?: boolean
  initialData?: Record<string, any>
  onSuccess?: () => void
  onCancel?: () => void
}

export function TaxDeclarationForm({
  employeeId,
  country,
  isEditing = false,
  initialData,
  onSuccess,
  onCancel,
}: TaxDeclarationFormProps) {
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [submitError, setSubmitError] = useState<string | null>(null)

  const countryConfig = getCountryConfig(country)
  const schema = getTaxDeclarationSchema(country)

  const form = useForm({
    resolver: zodResolver(schema),
    defaultValues: {
      taxIdType: countryConfig.taxIdType,
      ...initialData,
    },
  })

  const onSubmit = async (data: any) => {
    setIsSubmitting(true)
    setSubmitError(null)

    try {
      // Validate required fields for this country
      const missingFields = countryConfig.requiredTaxFields.filter(
        (field) => !data[field] && data[field] !== false && data[field] !== 0
      )

      if (missingFields.length > 0) {
        setSubmitError(`Missing required fields: ${missingFields.join(', ')}`)
        return
      }

      await updateEmployeeTaxInfo(employeeId, { country, ...data })
      toast.success('Tax information updated successfully')
      onSuccess?.()
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Failed to save tax information'
      setSubmitError(message)
      toast.error(message)
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <Card className="w-full">
      <CardHeader>
        <CardTitle>Tax Declaration - {countryConfig.name}</CardTitle>
        <CardDescription>
          {countryConfig.taxIdLabel} and other tax-related information for {countryConfig.name}
        </CardDescription>
      </CardHeader>
      <CardContent>
        {submitError && (
          <div className="mb-4 flex items-center gap-2 rounded-md bg-destructive/10 p-3 text-sm text-destructive">
            <AlertCircle className="h-4 w-4" />
            {submitError}
            <Button
              variant="ghost"
              size="sm"
              className="ml-auto"
              onClick={() => setSubmitError(null)}
            >
              Dismiss
            </Button>
          </div>
        )}

        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
            {/* Render country-specific form */}
            {country === 'australia' && <AustraliaTaxForm form={form} />}
            {(country === 'india' || country === 'nepal') && (
              <IndiaNepaliTaxForm form={form} country={country} />
            )}
            {country === 'uk' && <UKTaxForm form={form} />}
            {country === 'singapore' && <SingaporeTaxForm form={form} />}
            {country === 'newzealand' && <NewZealandTaxForm form={form} />}

            {/* Action Buttons */}
            <div className="flex justify-end gap-3 border-t pt-6">
              <Button
                type="button"

                disabled={isSubmitting}
