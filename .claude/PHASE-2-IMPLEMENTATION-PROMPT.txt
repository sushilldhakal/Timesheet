'use client'

import { useState } from 'react'
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Checkbox } from '@/components/ui/checkbox'
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card'
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from '@/components/ui/form'
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '@/components/ui/select'
import { Loader2, AlertCircle } from 'lucide-react'
import { toast } from 'sonner'
import { updateEmployeeTaxInfo } from '@/lib/api/employees'
import { taxInfoBodySchema } from '@/lib/validations/employee-payroll'
import type { z } from 'zod'

type TaxInfoFormData = z.infer<typeof taxInfoBodySchema>

interface TaxInfoFormProps {
  employeeId: string
  initialData?: {
    tfn?: string
    abn?: string
    superannuationFund?: string
    superannuationMemberNumber?: string
    taxWithholdingPercentage?: number
    helpDebt?: boolean
    salarySecureForm?: { url?: string; signedDate?: string }
  }
  onSuccess?: () => void
  onCancel?: () => void
}

export function TaxInfoForm({
  employeeId,
  initialData,
  onSuccess,
  onCancel,
}: TaxInfoFormProps) {
  const [isSubmitting, setIsSubmitting] = useState(false)
  const [submitError, setSubmitError] = useState<string | null>(null)

  const form = useForm<TaxInfoFormData>({
    resolver: zodResolver(taxInfoBodySchema),
    defaultValues: {
      tfn: initialData?.tfn || '',
      abn: initialData?.abn || '',
      superannuationFund: initialData?.superannuationFund || '',
      superannuationMemberNumber: initialData?.superannuationMemberNumber || '',
      taxWithholdingPercentage: initialData?.taxWithholdingPercentage || 37.5,
      helpDebt: initialData?.helpDebt || false,
    },
  })

  const onSubmit = async (data: TaxInfoFormData) => {
    setIsSubmitting(true)
    setSubmitError(null)

    try {
      await updateEmployeeTaxInfo(employeeId, data)
      toast.success('Tax information updated successfully')
      onSuccess?.()
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Failed to update tax information'
      setSubmitError(message)
      toast.error(message)
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <Card className="w-full">
      <CardHeader>
        <CardTitle>Tax Information</CardTitle>
        <CardDescription>Update employee tax details (admin only)</CardDescription>
      </CardHeader>
      <CardContent>
        {submitError && (
          <div className="mb-4 flex items-center gap-2 rounded-md bg-destructive/10 p-3 text-sm text-destructive">
            <AlertCircle className="h-4 w-4" />
            {submitError}
            <Button
              variant="ghost"
              size="sm"
              className="ml-auto"
              onClick={() => setSubmitError(null)}
            >
              Dismiss
            </Button>
          </div>
        )}

        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
            {/* Tax File Number */}
            <FormField
              control={form.control}
              name="tfn"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Tax File Number (TFN) *</FormLabel>
                  <FormControl>
                    <Input
                      placeholder="XXX-XXX-XXX"
                      {...field}
                      maxLength={11}
                      className="font-mono"
                    />
                  </FormControl>
                  <FormDescription>Format: XXX-XXX-XXX (11 characters)</FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />

            {/* ABN */}
            <FormField
              control={form.control}
              name="abn"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>ABN (for contractors only)</FormLabel>
                  <FormControl>
                    <Input
                      placeholder="XX-XXX-XXX-XXX"
                      {...field}
                      maxLength={11}
                      className="font-mono"
                    />
                  </FormControl>
                  <FormDescription>Leave blank if not a contractor</FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />

            {/* Superannuation Fund */}
            <FormField
              control={form.control}
              name="superannuationFund"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Superannuation Fund</FormLabel>
                  <FormControl>
                    <Input
                      placeholder="e.g., AustralianSuper"
                      {...field}
                    />
                  </FormControl>
                  <FormDescription>Name of the superannuation fund</FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />

            {/* Super Member Number */}
            <FormField
              control={form.control}
              name="superannuationMemberNumber"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Superannuation Member Number</FormLabel>
                  <FormControl>
                    <Input
                      placeholder="Member number from super fund"
                      {...field}
                      className="font-mono"
                    />
                  </FormControl>
                  <FormMessage />
                </FormItem>
              )}
            />

            {/* Tax Withholding */}
            <FormField
              control={form.control}
              name="taxWithholdingPercentage"
              render={({ field }) => (
                <FormItem>
                  <FormLabel>Tax Withholding %</FormLabel>
                  <FormControl>
                    <Input
                      type="number"
                      placeholder="37.5"
                      {...field}
                      min={0}
                      max={100}
                      step={0.1}
                      onChange={(e) => field.onChange(parseFloat(e.target.value))}
                      className="font-mono"
                    />
                  </FormControl>
                  <FormDescription>Percentage to withhold for taxes (0-100)</FormDescription>
                  <FormMessage />
                </FormItem>
              )}
            />

            {/* HELP/HECS Debt */}
            <FormField
              control={form.control}
              name="helpDebt"
              render={({ field }) => (
                <FormItem className="flex flex-row items-center justify-between rounded-lg border p-3">
                  <div className="space-y-0.5">
                    <FormLabel>HELP/HECS Debt</FormLabel>
                    <FormDescription>Employee has HELP/HECS debt</FormDescription>
                  </div>
                  <FormControl>
                    <Checkbox
                      checked={field.value}
                      onCheckedChange={field.onChange}
                    />
                  </FormControl>
                </FormItem>
              )}
            />

            {/* Info Box */}
            <div className="rounded-md bg-blue-50 p-3 text-sm text-blue-900">
              <p className="font-semibold">⚠️ Security Note</p>
              <p className="mt-1">All sensitive tax information is encrypted and securely stored. Only admins can view and edit.</p>
            </div>

            {/* Action Buttons */}
            <div className="flex justify-end gap-3">
              <Button
                type="button"
                variant="outline"
                onClick={onCancel}
                disabled={isSubmitting}
              >
                Cancel
              </Button>
              <Button
                type="submit"
                disabled={isSubmitting}
              >
                {isSubmitting && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
                {isSubmitting ? 'Saving...' : 'Save Tax Info'}
              </Button>
            </div>
          </form>
        </Form>
      </CardContent>
    </Card>
  )
}
