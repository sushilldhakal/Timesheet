import { differenceInDays } from 'date-fns'

export interface ComplianceStats {
  totalEmployees: number
  fullyCurrent: number
  warningCount: number
  criticalCount: number
  expiringPercentage: number
}

export interface ComplianceBreakdown {
  wwc: { compliant: number; warning: number; critical: number }
  policeClearance: { compliant: number; warning: number; critical: number }
  foodHandling: { compliant: number; warning: number; critical: number }
  induction: { completed: number; pending: number }
  codeOfConduct: { signed: number; pending: number }
}

export interface ComplianceIssue {
  employeeId: string
  employeeName: string
  issueType: string
  severity: 'critical' | 'warning' | 'info'
  description: string
  expiryDate?: string
  daysUntilExpiry?: number
}

export function calculateComplianceStats(
  employees: any[],
  complianceData: any[]
): ComplianceStats {
  const totalEmployees = employees.length

  let fullyCurrent = 0
  let warningCount = 0
  let criticalCount = 0

  employees.forEach((employee) => {
    const compliance = complianceData.find((c) => c.employeeId === employee._id)
    if (!compliance) return

    let hasIssue = false
    let hasCritical = false

    // Check each compliance item
    const statuses = [
      { status: compliance.wwcStatus, expiry: compliance.wwcExpiryDate },
      { status: compliance.policeClearanceStatus, expiry: compliance.policeClearanceExpiryDate },
      { status: compliance.foodHandlingStatus, expiry: compliance.foodHandlingExpiryDate },
    ]

    statuses.forEach(({ status, expiry }) => {
      if (status === 'expired' || status === 'pending') {
        hasCritical = true
        hasIssue = true
      } else if (expiry) {
        const daysLeft = differenceInDays(new Date(expiry), new Date())
        if (daysLeft < 0) {
          hasCritical = true
          hasIssue = true
        } else if (daysLeft < 30) {
          hasIssue = true
        }
      }
    })

    if (!hasIssue) {
      fullyCurrent++
    } else if (hasCritical) {
      criticalCount++
    } else {
      warningCount++
    }
  })

  return {
    totalEmployees,
    fullyCurrent,
    warningCount,
    criticalCount,
    expiringPercentage: totalEmployees > 0 ? Math.round(((warningCount + criticalCount) / totalEmployees) * 100) : 0,
  }
}

export function getComplianceBreakdown(complianceData: any[]): ComplianceBreakdown {
  return {
    wwc: complianceData.reduce(
      (acc, c) => {
        if (c.wwcStatus === 'cleared') acc.compliant++
        else if (c.wwcStatus === 'pending') acc.critical++
        else if (c.wwcExpiryDate && differenceInDays(new Date(c.wwcExpiryDate), new Date()) < 30)
          acc.warning++
        else acc.compliant++
        return acc
      },
      { compliant: 0, warning: 0, critical: 0 }
    ),
    policeClearance: complianceData.reduce(
      (acc, c) => {
        if (c.policeClearanceStatus === 'cleared') acc.compliant++
        else if (c.policeClearanceStatus === 'pending') acc.critical++
        else if (c.policeClearanceExpiryDate && differenceInDays(new Date(c.policeClearanceExpiryDate), new Date()) < 30)
          acc.warning++
        else acc.compliant++
        return acc
      },
      { compliant: 0, warning: 0, critical: 0 }
    ),
    foodHandling: complianceData.reduce(
      (acc, c) => {
        if (c.foodHandlingStatus === 'completed') acc.compliant++
        else if (c.foodHandlingStatus === 'pending') acc.critical++
        else if (c.foodHandlingExpiryDate && differenceInDays(new Date(c.foodHandlingExpiryDate), new Date()) < 30)
          acc.warning++
        else acc.compliant++
        return acc
      },
      { compliant: 0, warning: 0, critical: 0 }
    ),
    induction: complianceData.reduce(
      (acc, c) => {
        if (c.inductionCompleted) acc.completed++
        else acc.pending++
        return acc
      },
      { completed: 0, pending: 0 }
    ),
    codeOfConduct: complianceData.reduce(
      (acc, c) => {
        if (c.codeOfConductSigned) acc.signed++
        else acc.pending++
        return acc
      },
      { signed: 0, pending: 0 }
    ),
  }
}

export function getComplianceIssues(employees: any[], complianceData: any[]): ComplianceIssue[] {
  const issues: ComplianceIssue[] = []

  employees.forEach((employee) => {
    const compliance = complianceData.find((c) => c.employeeId === employee._id)
    if (!compliance) return

    const employeeName = `${employee.firstName} ${employee.lastName}`

    // Check WWC
    if (compliance.wwcStatus === 'expired') {
      issues.push({
        employeeId: employee._id,
        employeeName,
        issueType: 'WWC',
        severity: 'critical',
        description: 'WWC has expired',
        expiryDate: compliance.wwcExpiryDate,
      })
    } else if (compliance.wwcStatus === 'pending') {
      issues.push({
        employeeId: employee._id,
        employeeName,
        issueType: 'WWC',
        severity: 'critical',
        description: 'WWC pending',
      })
    } else if (compliance.wwcExpiryDate) {
      const daysLeft = differenceInDays(new Date(compliance.wwcExpiryDate), new Date())
      if (daysLeft < 0) {
        issues.push({
          employeeId: employee._id,
          employeeName,
          issueType: 'WWC',
          severity: 'critical',
          description: 'WWC has expired',
          expiryDate: compliance.wwcExpiryDate,
          daysUntilExpiry: daysLeft,
        })
      } else if (daysLeft < 30) {
        issues.push({
          employeeId: employee._id,
          employeeName,
          issueType: 'WWC',
          severity: 'warning',
          description: `WWC expiring in ${daysLeft} days`,
          expiryDate: compliance.wwcExpiryDate,
          daysUntilExpiry: daysLeft,
        })
      }
    }

    // Similar checks for other compliance items...
    // (Police Clearance, Food Handling, Induction, Code of Conduct)
  })

  return issues.sort((a, b) => {
    // Critical first, then by days until expiry
    if (a.severity !== b.severity) {
      return a.severity === 'critical' ? -1 : 1
    }
    if (a.daysUntilExpiry && b.daysUntilExpiry) {
      return a.daysUntilExpiry - b.daysUntilExpiry
    }
    return 0
  })
}

export function generateComplianceReport(

'use client'

import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card'
import { Badge } from '@/components/ui/badge'
import { eachDayOfInterval, startOfMonth, endOfMonth, format, isToday, differenceInDays } from 'date-fns'
import type { ComplianceIssue } from '@/lib/utils/compliance-dashboard'

interface ComplianceExpiryCalendarProps {
  issues: ComplianceIssue[]
}

export function ComplianceExpiryCalendar({ issues }: ComplianceExpiryCalendarProps) {
  const today = new Date()
  const monthStart = startOfMonth(today)
  const monthEnd = endOfMonth(today)
  const daysInMonth = eachDayOfInterval({ start: monthStart, end: monthEnd })

  // Group issues by expiry date
  const issuesByDate = new Map<string, ComplianceIssue[]>()
  issues.forEach((issue) => {
    if (issue.expiryDate) {
      const dateKey = format(new Date(issue.expiryDate), 'yyyy-MM-dd')
      if (!issuesByDate.has(dateKey)) {
        issuesByDate.set(dateKey, [])
      }
      issuesByDate.get(dateKey)!.push(issue)
    }
  })

  const weekDays = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat']

  return (
    <Card>
      <CardHeader>
        <CardTitle className="text-base">Upcoming Expirations</CardTitle>
        <CardDescription>{format(today, 'MMMM yyyy')}</CardDescription>
      </CardHeader>
      <CardContent className="space-y-6">
        {/* Calendar Grid */}
        <div>
          <div className="grid grid-cols-7 gap-2 mb-2">
            {weekDays.map((day) => (
              <div key={day} className="text-center text-xs font-semibold text-gray-600 py-2">
                {day}
              </div>
            ))}
          </div>
          <div className="grid grid-cols-7 gap-2">
            {daysInMonth.map((date) => {
              const dateKey = format(date, 'yyyy-MM-dd')
              const dayIssues = issuesByDate.get(dateKey) || []
              const isTodayDate = isToday(date)

              return (
                <div
                  key={dateKey}
                  className={`p-2 rounded-md text-sm text-center relative border ${
                    isTodayDate
                      ? 'bg-blue-100 border-blue-300'
                      : dayIssues.length > 0
                        ? 'bg-red-50 border-red-200'
                        : 'bg-gray-50 border-gray-200'
                  }`}
                >
                  <div className="font-semibold">{date.getDate()}</div>
                  {dayIssues.length > 0 && (
                    <div className="text-xs text-red-600 font-semibold">{dayIssues.length} exp.</div>
                  )}
                </div>
              )
            })}
          </div>
        </div>

        {/* Upcoming Expirations List */}
        <div className="space-y-3 border-t pt-4">
          <h4 className="font-semibold text-sm">Next 10 Expirations</h4>
          {Array.from(issuesByDate.entries())
            .filter(([dateKey]) => {
              const date = new Date(dateKey)
              return differenceInDays(date, today) >= 0 && differenceInDays(date, today) <= 30
            })
            .sort((a, b) => new Date(a[0]).getTime() - new Date(b[0]).getTime())
            .slice(0, 10)
            .map(([dateKey, dayIssues]) => (
              <div key={dateKey} className="p-3 bg-gray-50 rounded-md">
                <div className="flex items-center justify-between mb-2">
                  <p className="text-sm font-semibold">{format(new Date(dateKey), 'MMM d, yyyy')}</p>
                  <Badge variant={dayIssues[0].severity === 'critical' ? 'destructive' : 'secondary'}>
                    {dayIssues.length} {dayIssues.length === 1 ? 'issue' : 'issues'}
                  </Badge>
                </div>
                <ul className="text-xs space-y-1">
                  {dayIssues.map((issue, idx) => (
                    <li key={idx} className="text-gray-600">
                      • {issue.employeeName} - {issue.issueType}
                    </li>
                  ))}
                </ul>
              </div>
            ))}
        </div>
      </CardContent>
    </Card>
  )
}

    if (!employees) return null
    return calculateComplianceStats(employees, complianceData)
  }, [employees, complianceData])

  const breakdown = useMemo(() => {
    return getComplianceBreakdown(complianceData)
  }, [complianceData])

  const allIssues = useMemo(() => {
    if (!employees) return []
    return getComplianceIssues(employees, complianceData)
  }, [employees, complianceData])

  // Filter issues
  const filteredIssues = useMemo(() => {
    return allIssues.filter((issue) => {
      if (severityFilter !== 'all' && issue.severity !== severityFilter) return false
      if (issueTypeFilter !== 'all' && issue.issueType !== issueTypeFilter) return false
      return true
    })
  }, [allIssues, severityFilter, issueTypeFilter])

  const handleExportReport = () => {
    if (!stats) return

    const report = generateComplianceReport(stats, breakdown, allIssues)

    // Create blob and download
    const blob = new Blob([report], { type: 'text/plain' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `compliance-report-${new Date().toISOString().split('T')[0]}.txt`
    document.body.appendChild(a)
    a.click()
    document.body.removeChild(a)
    URL.revokeObjectURL(url)

    toast.success('Report downloaded successfully')
  }

  const handleRefresh = () => {
    // Trigger refetch
    toast.info('Refreshing compliance data...')
  }

  const issueTypes = [...new Set(allIssues.map((i) => i.issueType))]

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-3xl font-bold">Compliance Dashboard</h1>
          <p className="text-gray-600 mt-1">Organization-wide compliance monitoring and reporting</p>
        </div>
        <div className="flex gap-2">
          <Button
            variant="outline"
            onClick={handleRefresh}
            disabled={employeesLoading}
          >
            <RefreshCw className="mr-2 h-4 w-4" />
            Refresh
          </Button>
          <Button onClick={handleExportReport} disabled={!stats}>
            <Download className="mr-2 h-4 w-4" />
            Export Report
          </Button>
        </div>
      </div>

      {/* Overview */}
      {stats && (
        <ComplianceOverviewCard
          totalEmployees={stats.totalEmployees}
          fullyCurrent={stats.fullyCurrent}
          warningCount={stats.warningCount}
          criticalCount={stats.criticalCount}
          expiringPercentage={stats.expiringPercentage}
        />
      )}

      {/* Breakdown and Calendar */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        <ComplianceBreakdownCard breakdown={breakdown} />
        <ComplianceExpiryCalendar issues={allIssues} />
      </div>

      {/* Issues Table with Filters */}
      <Card>
        <CardHeader>
          <CardTitle>Filter Issues</CardTitle>
        </CardHeader>
        <CardContent className="flex gap-4 flex-wrap">
          <div className="flex-1 min-w-48">
            <label className="text-sm font-medium">Severity</label>
            <Select value={severityFilter} onValueChange={(v) => setSeverityFilter(v as any)}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All Severities</SelectItem>
                <SelectItem value="critical">Critical Only</SelectItem>
                <SelectItem value="warning">Warnings Only</SelectItem>
              </SelectContent>
            </Select>
          </div>

          <div className="flex-1 min-w-48">
            <label className="text-sm font-medium">Issue Type</label>
            <Select value={issueTypeFilter} onValueChange={setIssueTypeFilter}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All Types</SelectItem>
                {issueTypes.map((type) => (
                  <SelectItem key={type} value={type}>
                    {type}
                  </SelectItem>
                ))}
              </SelectContent>
            </Select>
          </div>
        </CardContent>
      </Card>

      {/* Issues Table */}
      <ComplianceIssuesTable issues={filteredIssues} showAll={true} />
    </div>
  )
}
